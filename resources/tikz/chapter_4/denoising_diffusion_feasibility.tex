% usage: \drawBarGroup{800}{34/blue,52/red,41/green}
\newcommand{\drawBarGroup}[2]{%
  % count bars
  \def\nBars{0}%
  \foreach \val/\clr [count=\k] in {#2}{%
    \xdef\nBars{\k}%
  }%

  % group geometry
  \pgfmathsetmacro{\GroupWidth}{\nBars*\BarWidth + (\nBars-1)*\BarInnerGap}%
  \pgfmathsetmacro{\GroupStart}{#1 - 0.5*\GroupWidth}%

  % draw bars
  \foreach \val/\clr [count=\k from 0] in {#2}{%
    \pgfmathsetmacro{\XLeft}{\GroupStart + \k*(\BarWidth+\BarInnerGap)}%
    \pgfmathsetmacro{\XRight}{\XLeft + \BarWidth}%
    \edef\BarPath{%
      \noexpand\path[draw=none,fill=\clr]%
      (axis cs:\XLeft,0) rectangle (axis cs:\XRight,\val);%
    }%
    \BarPath
  }%
}
