\begin{tikzpicture}[

  edge/.style={-{Latex[length=2mm,width=1.4mm]}, line width=0.55pt, draw=blue!50!black},
  inputneuron/.style={circle, draw=blue!55!black, fill=blue!15, line width=0.8pt, minimum size=7.5mm, inner sep=0pt},
  hiddenneuron/.style={circle, draw=green!50!black, fill=green!25, line width=0.8pt, minimum size=7.5mm, inner sep=0pt},

% -------------------------------------------------
% Helper: centered vertical stack of neurons
% #1 = count, #2 = x-coordinate, #3 = name prefix, #4 = node style
% -------------------------------------------------
\newcommand{\stacknodes}[4]{%
  \pgfmathsetmacro{\N}{#1}
  \foreach \i in {1,...,#1} {%
    \node[#4] (#3\i) at (#2,{-(\i-1)*\ysep + 0.5*(\N-1)*\ysep}) {};
  }%
}

% -------------------------------------------------
% Layer x-coordinates and fixed label/parameter rows
% -------------------------------------------------
\coordinate (XInput)  at (0,0);
\coordinate (XHiddenA) at (\xsep,0);
\coordinate (XHiddenB) at (2*\xsep,0);
\coordinate (XOutput) at (3*\xsep,0);
\def\labelY{3.45}
\def\paramY{-2.85}

% -------------------------------------------------
% Layers
% -------------------------------------------------
\stacknodes{\nI}{0}{I}{inputneuron}
\node[bias] (b0) at (0,2.15) {$1$};

\stacknodes{\nH}{\xsep}{H}{hiddenneuron}
\node[bias] (b1) at (\xsep,2.65) {$1$};

\stacknodes{\nHtwo}{2*\xsep}{G}{hiddenneuron}
\node[bias] (b2) at (2*\xsep,2.20) {$1$};

\stacknodes{\nO}{3*\xsep}{O}{outputneuron}

% -------------------------------------------------
% Dense connections
% -------------------------------------------------
\foreach \i in {1,...,\nI} {
  \foreach \j in {1,...,\nH} {
    \draw[edge] (I\i) -- (H\j);
  }
}
\foreach \j in {1,...,\nH} {
  \draw[edge] (b0) -- (H\j);
}

\foreach \i in {1,...,\nH} {
  \foreach \j in {1,...,\nHtwo} {
    \draw[edge] (H\i) -- (G\j);
  }
}
\foreach \j in {1,...,\nHtwo} {
  \draw[edge] (b1) -- (G\j);
}

\foreach \i in {1,...,\nHtwo} {
  \foreach \j in {1,...,\nO} {
    \draw[edge] (G\i) -- (O\j);
  }
}
\foreach \j in {1,...,\nO} {
  \draw[edge] (b2) -- (O\j);
}

% -------------------------------------------------
% Aligned layer labels: all on the same horizontal row
% -------------------------------------------------
\node[font=\bfseries, text=TitleColor, align=center] at (0,\labelY)
  {Input layer\\[-1pt]$x$};
\node[font=\bfseries, text=TitleColor, align=center] at (\xsep,\labelY)
  {Hidden 1\\[-1pt]$\sigma\!\left(W^{(1)}x+b^{(1)}\right)$};
\node[font=\bfseries, text=TitleColor, align=center] at (2*\xsep,\labelY)
  {Hidden 2\\[-1pt]$\sigma\!\left(W^{(2)}h^{(1)}+b^{(2)}\right)$};
\node[font=\bfseries, text=TitleColor, align=center] at (3*\xsep,\labelY)
  {Output\\[-1pt]$\hat y=f_{\theta}(x)$};

% -------------------------------------------------
% Aligned parameter labels: all on the same horizontal row
% -------------------------------------------------
\node[font=\bfseries, text=TitleColor] at (0.5*\xsep,\paramY) {$W^{(1)},\ b^{(1)}$};
\node[font=\bfseries, text=TitleColor] at (1.5*\xsep,\paramY) {$W^{(2)},\ b^{(2)}$};
\node[font=\bfseries, text=TitleColor] at (2.5*\xsep,\paramY) {$W^{(3)},\ b^{(3)}$};
